#include<stdio.h>
int main()
{
 int m,n,p,i,j,k;
 float u1[5][5],u2[5][5],u3[5][5],c[10],max;
 printf("Enter the how many rows:\n");
 scanf("%d",&m);
 printf("Enter the how many cols:\n");
 scanf("%d",&n);
 printf("Eneter the entries in fuzzy set A:\n");
 for(i=1;i<=m;i++)
 {
   for(j=1;j<=n;j++)
   {
     scanf("%f",&u1[i][j]);
   }
 }
 printf("Enter the entries in fuzzy set B :\n");
 for(i=1;i<=m;i++)
 {
   for(j=1;j<=n;j++)
   {
     scanf("%f",&u2[i][j]);
   }
 }
//performing max product composition
 for(i=1;i<=m;i++)
 {
   for(j=1;j<=n;j++)
   {
    for(k=1;k<=n;k++)
    {
      c[k]=u1[i][k]*u2[k][j];
    }
    max=c[1];
    for(k=2;k<=n;k++)
    {
       if(c[k]>max)
           max=c[k];
    }
    u3[i][j]=max;
  }
}
printf("\n The max min composition of fuzzy set A and B is:\n");
for(i=1;i<=m;i++)
{
  for(j=1;j<=n;j++)
  {
     printf("%.2f\t",u3[i][j]);
  }
  printf("\n");
 }

 return 0;
}
